\def\TitlePage{

  \setuppagenumber[state=stop]

  \framed
  [frame=off,offset=overlay,width=fit,height=.2\textheight,option=empty]
  {}

  \framed
  [frame=off,width=\titlewidth,height=broad,align=middle]
  {\switchtobodyfont[20pt,sans]
    \bf \DocumentType }

  \blank[big]

  \framed
  [frame=off,width=\titlewidth,height=broad,align=middle]
  {\switchtobodyfont[18pt]
   \it \Title}

  \blank[2*big]

  \framed
  [frame=off,width=\titlewidth,height=broad,align=middle]
  {\hl[8]}

  \blank[2*big]

  \framed
  [frame=off,width=\titlewidth,height=broad,align=middle]
  {\ss\bfb\SubTitle}

  \framed
  [frame=off,width=\titlewidth,height=broad,align=middle]
  {\ss\bfb\DocumentDate}

  \page[yes,reset]
  \setuppagenumber[state=start]

}
